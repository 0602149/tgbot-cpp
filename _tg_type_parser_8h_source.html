334 template<typename T>
335 std::shared_ptr<T> tryParseJson(JsonToTgTypeFunc<T> parseFunc, const boost::property_tree::ptree& data, const std::string& keyName) const {
336 auto treeItem = data.find(keyName);
337 if (treeItem == data.not_found()) {
338 return std::shared_ptr<T>();
339 }
340 return (this->*parseFunc)(treeItem->second);

399 template<typename T>
400 std::string parseArray(TgTypeToJsonFunc<T> parseFunc, const std::vector<std::shared_ptr<T>>& objects) const {
401 if (objects.empty())
402 return "";
403 std::string result;
404 result += '[';
405 for (const std::shared_ptr<T>& item : objects) {
406 result += (this->*parseFunc)(item);
407 result += ',';
408 }
409 result.erase(result.length() - 1);
410 result += ']';
411 return result;
412 }
